    allCameraData = [getFrameData(), runThermalFilter()]
    data_set_ofInfMech = inferenceMechanism() //reply tokens
    dataToSave = [allCameraData, data_set_ofInfMech]

    //color concentration calculations
    sequesteredColorConcentrationCamData = [allCameraData[1][0], allCameraData[1][1], allCameraData[1][2]];
    red = (sequesteredColorConcentrationCamData[0]/(sum(sequesteredColorConcentrationCamData)) )
    green = (sequesteredColorConcentrationCamData[1]/(sum(sequesteredColorConcentrationCamData)) )
    blue = (sequesteredColorConcentrationCamData[2]/(sum(sequesteredColorConcentrationCamData)) )
        
    //color collection
    colorConcentrationCamData = {
        Red:red;
        Green:green;
        Blue:blue;
    };
    sequesteredInfData = {
        userPrompt:data_set_ofInfMech[0];
        WebLLMResponse:data_set_ofInfMech[1];        
    };
    dataSequesteration = {
        Prompts:sequesteredInfData[userPrompt]
        Inferences:sequesteredInfData[WebLLMResponse];
        Tokens:data_set_ofInfMech[0];
        colorConcentrations:colorConcentrationCamData[Blue];
    };
    x = dataSequesteration[Prompts];
    y = dataSequesteration[Inferences];
    z = dataSequesteration[Tokens];
    w = dataSequesteration[colorConcentrations]
    formattingOfSequesteration(X,Y,Z,W) {
        //formatting of rows and columns
        columns_aka_yAxis = ["User Prompts", "Inferences","Token Usage", "Blue Concentration"]
        rows_aka_xAxis = [" "]
        rows_aka_xAxis.push(X)
        rows_aka_xAxis.push(Y)
        rows_aka_xAxis.push(Z)
        rows_aka_xAxis.push(W)   
        xy = [columns_aka_yAxis, rows_aka_xAxis]
        return xy
    }
----SET/PROCESSING----
